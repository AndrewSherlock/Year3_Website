namespace App\Entity;

use Doctrine\ORM\Mapping as ORM;
use Symfony\Component\Security\Core\User\UserInterface;


/**
 * User class
 * @ORM\Entity(repositoryClass="App\Repository\UserRepository")
 * Class User
 * @package App\Entity
 */
class User implements  UserInterface, \Serializable
{
    /**
     *  id for user
     * @ORM\Id
     * @ORM\GeneratedValue
     * @ORM\Column(type="integer")
     * @var integer
     */
    private $id;

    /**
     * name of the user
     * @ORM\Column(type="string")
     * @var string
     */
    private $username;

    /**
     * password for user
     * @ORM\Column(type="string", length=64)
     * @var string
     */
    private $password;

    /**
     * User roles
     * @ORM\Column(type="json_array")
     * @var array
     */
    private $roles = [];


    /**
     * the foods the user created
     * @ORM\OneToMany(targetEntity="App\Entity\Food", mappedBy="user")
     * @var Food
     */
    private $foods;

    /**
     * Reviews that the user has given
     * @ORM\OneToMany(targetEntity="App\Entity\Review", mappedBy="user")
     * @var Review
     */
    private $reviews;

    /**
     * User votes
     * @ORM\OneToMany(targetEntity="App\Entity\Vote", mappedBy="user")
     * @var Vote
     */
    private $votes;

    /**
     * get the user id
     * @return int
     */
    public function getId()
    {
        return $this->id;
    }

    /**
     * get the user name
     * @return string
     */
    public function getUsername()
    {
        return $this->username;
    }

    /**
     * set the user name
     * @param string $username
     */
    public function setUsername($username): void
    {
        $this->username = $username;
    }

    /**
     * get the user password
     * @return string
     */
    public function getPassword()
    {
        return $this->password;
    }

    /**
     * set the user password
     * @param string $password
     */
    public function setPassword($password): void
    {
        $this->password = $password;
    }

    /**
     * set the array serialized
     * @return array
     */
    public function serialize()
    {
        return serialize(array( $this->id, $this->username, $this->password));
    }

    /**
     * unserilized the user information
     * @param string $serialized
     */
    public function unserialize($serialized)
    {
        list ( $this->id, $this->username, $this->password ) = unserialize($serialized);
    }

    /**
     * gets the user roles
     * @return array
     */
    public function getRoles()
    {
        $roles = $this->roles;
        return array_unique($roles);
    }

    /**
     * sets the users roles
     * @param $roles
     * @return $this
     */
    public function setRoles($roles)
    {
        $this->roles = $roles;

        return $this;
    }

    /**
     * unused in this. we use bcrypt
     * @return null|string
     */
    public function getSalt()
    {
        // TODO: Implement getSalt() method.

        return null;
    }

    /**
     * not implemented
     */
    public function eraseCredentials()
    {
        // TODO: Implement eraseCredentials() method.
    }
}
